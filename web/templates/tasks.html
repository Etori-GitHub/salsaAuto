function toggleSidebar() { document.getElementById('sidebar').classList.toggle('collapsed'); } async function loadTasks() { const data = await fetchAPI('/api/tasks/list'); tasks = data.tasks || []; renderTaskList(); } function renderTaskList() { const container = document.getElementById('task-list'); if (tasks.length === 0) { container.innerHTML = '

暂无任务，请先在计算器中创建任务

'; return; } container.innerHTML = tasks.map(task => `
${task.task_name}
${task.pay_type_name}
${task.store_name} ${task.date_range?.start || '-'} ~ ${task.date_range?.end || '-'} Y${(task.total_amount || 0).toFixed(2)}
`).join(''); } async function selectTask(taskId) { // 更新选中状态 document.querySelectorAll('.task-card').forEach(el => { el.classList.remove('selected'); }); document.querySelector(`.task-card[data-id="${taskId}"]`)?.classList.add('selected'); // 获取任务详情 const data = await fetchAPI(`/api/tasks/${taskId}`); if (data.success) { selectedTask = data.task; renderTaskDetail(); } else { alert('获取任务详情失败: ' + data.message); } } function renderTaskDetail() { if (!selectedTask) return; const card = document.getElementById('task-detail-card'); const content = document.getElementById('task-detail-content'); const days = selectedTask.days || []; content.innerHTML = `
门店：${selectedTask.store_name} (ID: ${selectedTask.store_id})
支付方式：${selectedTask.pay_type_name}
总金额：Y${(selectedTask.total_amount || 0).toFixed(2)}
创建时间：${selectedTask.created_at || '-'}
${days.map(day => `
${day.date}
${(day.dishes || []).map(d => ` ${d.name} x${d.quantity} `).join('')}
Y${(day.total_amount || 0).toFixed(2)}
`).join('')}
`; card.style.display = 'block'; document.getElementById('execute-card').style.display = 'block'; } async function executeTask() { if (!selectedTask) { alert('请先选择任务'); return; } const btn = document.getElementById('execute-btn'); btn.disabled = true; btn.innerHTML = ' 执行中...'; const memberType = document.getElementById('member-type').value; const formData = new FormData(); formData.append('member_type', memberType); try { const result = await fetchAPI(`/api/tasks/${selectedTask.task_id}/execute`, { method: 'POST', body: formData }); if (result.success) { alert(result.message); } else { alert('执行失败: ' + result.message); } } catch (e) { alert('执行失败: ' + e.message); } btn.disabled = false; btn.innerHTML = ' 开始执行'; }
